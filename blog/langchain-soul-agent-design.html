从技术原理到代码实现，让 AI Agent 拥有稳定的身份和持续的记忆

一、问题：为什么 AI Agent 需要"人格"？

用 LangChain 构建过 Agent 的开发者，大概都遇到过这个问题：

每次对话，AI 都是"全新"的。

它不记得你是谁，不记得上次聊了什么，更没有稳定的"性格"。

这就像每次打电话给客服，对面都是不同的人——你得重新解释背景，重新建立信任。

而 OpenClaw 的设计很聪明：它用一套人格文件来解决这个问题：

文件 作用 类比
SOUL.md 定义 Agent 的身份、性格、价值观 人的灵魂/性格
USER.md 描述用户是谁、偏好、背景 对用户的了解
MEMORY.md 存储重要记忆、决策、学习 长期记忆
AGENTS.md 定义行为规范、工作流程 工作守则

这套机制的核心思想是：用文件持久化人格和记忆，每次启动时加载。

1.1 这个方案的好处

  • 人格稳定：每次对话，AI 都是"同一个人"
  • 记忆延续：知道上次聊了什么，记得你的偏好
  • 可编辑：改文件就能调整人格，无需改代码
  • 可移植：把文件夹复制到另一台机器，人格跟着走
  • 可版本控制：用 Git 追踪人格演化历史

3.2 核心代码实现

soul_agent.py - 核心类定义
"""
SoulAgent: 基于 LangChain 的人格持久化 Agent
模仿 OpenClaw 的 SOUL.md 机制
"""

from pathlib import Path
from typing import Optional, List
from datetime import datetime

from langchain.chat_models import ChatOpenAI
from langchain.schema import SystemMessage, HumanMessage, AIMessage
from langchain.agents import AgentExecutor, create_openai_functions_agent
from langchain.tools import Tool
from langchain.memory import ConversationBufferMemory
from langchain.prompts import ChatPromptTemplate, MessagesPlaceholder


class SoulAgent:
    """
    有人格的持久化 Agent
    
    核心机制：
    1. 启动时加载 SOUL.md / USER.md / MEMORY.md
    2. 拼接成 System Prompt 注入 LLM
    3. 对话结束后提取重要信息，更新 MEMORY.md
    """
    
    def __init__(
        self,
        workspace: str,
        model: str = "gpt-4",
        tools: Optional[List[Tool]] = None,
    ):
        """
        初始化 SoulAgent
        
        Args:
            workspace: 工作目录，存放 SOUL.md 等文件
            model: 使用的模型名称
            tools: Agent 可用的工具列表
        """
        self.workspace = Path(workspace)
        self.tools = tools or []
        
        # 加载人格文件
        self.soul = self._load_file("SOUL.md")
        self.user = self._load_file("USER.md")
        self.memory = self._load_file("MEMORY.md")
        self.agents = self._load_file("AGENTS.md")
        
        # 初始化 LLM
        self.llm = ChatOpenAI(model=model, temperature=0.7)
        
        # 对话记忆（短期）
        self.conversation_memory = ConversationBufferMemory(
            memory_key="chat_history",
            return_messages=True
        )
        
        # 构建 Agent
        self.agent_executor = self._build_agent()
    
    def _load_file(self, filename: str) -> str:
        """加载人格文件"""
        filepath = self.workspace / filename
        if filepath.exists():
            return filepath.read_text(encoding="utf-8")
        return ""
    
    def _save_file(self, filename: str, content: str):
        """保存文件"""
        filepath = self.workspace / filename
        filepath.write_text(content, encoding="utf-8")
    
    def build_system_prompt(self) -> str:
        """
        构建 System Prompt
        
        这是最关键的方法：把人格文件拼接成完整的 System Prompt
        """
        prompt_parts = []
        
        # 1. 身份定义（SOUL.md）
        if self.soul:
            prompt_parts.append(f"""# 你的身份

{self.soul}

---
*以上定义了你是谁，你的性格、价值观和行为准则。*
""")
        
        # 2. 用户信息（USER.md）
        if self.user:
            prompt_parts.append(f"""# 你服务的人

{self.user}

---
*以上是你服务对象的信息，记住这些以便提供个性化服务。*
""")
        
        # 3. 长期记忆（MEMORY.md）
        if self.memory:
            prompt_parts.append(f"""# 你的记忆

{self.memory}

---
*以上是你的重要记忆，在回答问题时参考这些信息。*
""")
        
        # 4. 行为规范（AGENTS.md）
        if self.agents:
            prompt_parts.append(f"""# 你的行为规范

{self.agents}

---
*以上是你必须遵守的行为准则和工作流程。*
""")
        
        # 5. 核心指令
        prompt_parts.append("""# 核心指令

1. 始终保持你的身份和性格
2. 记住用户的偏好和历史对话
3. 在合适的时候更新你的记忆
4. 对用户保持真诚和有帮助
""")
        
        return "\n\n".join(prompt_parts)
    
    def _build_agent(self) -> AgentExecutor:
        """构建 LangChain Agent"""
        
        # 创建 Prompt 模板
        prompt = ChatPromptTemplate.from_messages([
            ("system", "{system_prompt}"),
            MessagesPlaceholder(variable_name="chat_history", optional=True),
            ("human", "{input}"),
            MessagesPlaceholder(variable_name="agent_scratchpad"),
        ])
        
        # 创建 Agent
        agent = create_openai_functions_agent(
            llm=self.llm,
            tools=self.tools,
            prompt=prompt
        )
        
        # 创建 AgentExecutor
        return AgentExecutor(
            agent=agent,
            tools=self.tools,
            memory=self.conversation_memory,
            verbose=True,
            handle_parsing_errors=True
        )
    
    def chat(self, message: str) -> str:
        """
        对话入口
        
        Args:
            message: 用户消息
        
        Returns:
            AI 的回复
        """
        # 构建 System Prompt
        system_prompt = self.build_system_prompt()
        
        # 调用 Agent
        response = self.agent_executor.invoke({
            "input": message,
            "system_prompt": system_prompt
        })
        
        return response["output"]
    
    def save_memory(self, new_memory: str):
        """
        保存新的记忆到 MEMORY.md
        
        Args:
            new_memory: 要保存的新记忆内容
        """
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M")
        
        # 追加到现有记忆
        updated_memory = self.memory + f"\n\n## {timestamp}\n{new_memory}\n"
        
        # 保存到文件
        self._save_file("MEMORY.md", updated_memory)
        self.memory = updated_memory
    
    def update_user_info(self, info: str):
        """更新用户信息"""
        self._save_file("USER.md", info)
        self.user = info
    
    def reload(self):
        """重新加载所有人格文件（热更新）"""
        self.soul = self._load_file("SOUL.md")
        self.user = self._load_file("USER.md")
        self.memory = self._load_file("MEMORY.md")
        self.agents = self._load_file("AGENTS.md")

四、进阶：记忆管理和优化

4.1 记忆增长问题

随着对话越来越多，MEMORY.md 会越来越大，导致：

  • Token 消耗增加
  • 加载速度变慢
  • 重要信息被淹没

解决方案：分层记忆架构

memory_manager.py - 记忆管理
"""
记忆管理：解决记忆增长问题
"""

from typing import List, Dict
from langchain.embeddings import OpenAIEmbeddings
from langchain.vectorstores import Chroma
from langchain.text_splitter import RecursiveCharacterTextSplitter


class MemoryManager:
    """
    分层记忆管理
    
    架构：
    - 热记忆：最近 7 天，直接加载到 prompt
    - 温记忆：7-30 天，压缩摘要后加载
    - 冷记忆：30 天以上，存入向量库，按需检索
    """
    
    def __init__(self, workspace: str):
        self.workspace = Path(workspace)
        self.embeddings = OpenAIEmbeddings()
        self.vectorstore = Chroma(
            persist_directory=str(self.workspace / "vectorstore"),
            embedding_function=self.embeddings
        )
    
    def add_memory(self, content: str, metadata: dict):
        """添加记忆"""
        # 1. 写入今日记忆文件
        today = datetime.now().strftime("%Y-%m-%d")
        memory_file = self.workspace / "memory" / f"{today}.md"
        
        with open(memory_file, "a", encoding="utf-8") as f:
            f.write(f"\n{content}\n")
        
        # 2. 同时存入向量库
        self.vectorstore.add_texts(
            texts=[content],
            metadatas=[metadata]
        )

        results = self.vectorstore.similarity_search(query, k=k)

4.2 自动记忆提取

auto_memory.py - 自动提取记忆
"""
对话结束后自动提取重要信息，写入 MEMORY.md
"""

from langchain.chat_models import ChatOpenAI
from langchain.schema import HumanMessage, SystemMessage


class AutoMemoryExtractor:
    """自动从对话中提取重要信息"""
    
    def __init__(self, model: str = "gpt-3.5-turbo"):
        self.llm = ChatOpenAI(model=model, temperature=0)
    
    def extract(self, conversation: str) -> dict:
        """
        从对话中提取重要信息
        
        Returns:
            {
                "new_facts": ["用户叫刘彬", ...],
                "decisions": ["决定使用 LangChain", ...],
                "todos": ["明天发布博客", ...],
                "preferences": ["用户喜欢简洁的回答", ...]
            }
        """
        prompt = f"""分析以下对话，提取重要信息：

{conversation}

请以 JSON 格式返回：
{{
    "new_facts": ["新发现的事实", ...],
    "decisions": ["做出的决策", ...],
    "todos": ["待办事项", ...],
    "preferences": ["用户偏好", ...]
}}

只提取真正重要的信息，忽略闲聊内容。"""
        
        response = self.llm.invoke([
            SystemMessage(content="你是一个信息提取专家"),
            HumanMessage(content=prompt)
        ])
        
        # 解析 JSON
        import json
        return json.loads(response.content)
    
    def format_for_memory(self, extracted: dict) -> str:
        """格式化为 MEMORY.md 格式"""
        parts = []
        
        if extracted.get("new_facts"):
            parts.append("### 新发现")
            for fact in extracted["new_facts"]:
                parts.append(f"- {fact}")
        
        if extracted.get("decisions"):
            parts.append("### 决策")
            for decision in extracted["decisions"]:
                parts.append(f"- {decision}")
        
        if extracted.get("todos"):
            parts.append("### 待办")
            for todo in extracted["todos"]:
                parts.append(f"- [ ] {todo}")
        
        return "\n".join(parts)


# 集成到 SoulAgent
class SoulAgentWithAutoMemory(SoulAgent):
    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.memory_extractor = AutoMemoryExtractor()
    
    def chat(self, message: str) -> str:
        response = super().chat(message)
        
        # 对话结束后，自动提取记忆
        # （实际应用中可以异步执行）
        extracted = self.memory_extractor.extract(
            f"用户: {message}\nAI: {response}"
        )
        
        if any(extracted.values()):  # 有重要信息
            memory_text = self.memory_extractor.format_for_memory(extracted)
            self.save_memory(memory_text)
        
        return response

核心思想一致：用文件持久化人格和记忆，每次启动时加载。

OpenClaw 是完整的框架，SoulAgent 是可以在 LangChain 中复用的核心机制。

七、总结

通过模仿 OpenClaw 的 SOUL.md 机制，我们可以在 LangChain 中构建有人格、有记忆、可持久化的 AI Agent。

核心要点：

1. 人格文件：SOUL.md / USER.md / MEMORY.md / AGENTS.md

2. 加载机制：启动时读取文件，拼接到 System Prompt

3. 记忆管理：分层存储 + 自动提取 + 定期压缩

4. 热更新：修改文件即可调整人格，无需改代码

这套方案让 AI Agent 从"每次对话都是陌生人"，变成"每次对话都是老朋友"——这才是真正的 AI 助手该有的样子。
