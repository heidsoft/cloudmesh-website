一、什么是 Harness 架构？

在 AI Agent 领域，Harness 指的是一套用于运行、管理和编排 AI Agent 的基础设施框架。

核心问题：如何让多个 AI Agent 协同工作，支持不同的运行时（Claude Code、Codex、Gemini CLI），并且能够持久化会话、绑定线程？

OpenClaw 的 Harness 架构提供了答案：

1.1 Harness 要解决的核心问题

问题 Harness 解决方案
多种 Agent 运行时共存 Runtime Router 统一抽象
会话需要持久化 Session Store + JSONL 日志
对话要绑定到线程 Thread Binding 机制
子 Agent 需要隔离 Session Key 命名空间
长时间任务不阻塞 异步执行 + 事件通知

二、核心概念解析

2.1 Session（会话）

Session 是 Harness 的核心抽象，代表一个 Agent 的运行实例。

Session Key 命名规范
// Session Key 格式
agent:<agentId>:<runtime>:<uuid>

// 示例
agent:main:subagent:a1b2c3d4-e5f6-7890-abcd-ef1234567890
agent:codex:acp:fedcba09-8765-4321-fedc-ba0987654321
agent:claude:acp:12345678-1234-5678-1234-567812345678

Session 包含：

  • Session Key：唯一标识符
  • Session File：持久化文件（JSONL 格式）
  • Thread Binding：绑定的线程信息
  • Runtime：运行时类型（acp/subagent）
  • State：当前状态（active/idle/closed）

2.2 Runtime（运行时）

Harness 支持两种运行时：

运行时 说明 使用场景
subagent OpenClaw 原生子代理 轻量级任务、需要沙箱隔离
acp ACP 协议外部 Harness Claude Code、Codex、Gemini CLI

Thread Binding 是 Harness 的杀手级特性：

3.2 Session Manager 设计

session_manager.py - 核心实现
"""
Session Manager: 会话生命周期管理
"""

from dataclasses import dataclass, field
from enum import Enum
from typing import Dict, Optional, List, Callable, Any
from datetime import datetime
from pathlib import Path
import uuid
import json


class RuntimeType(Enum):
    SUBAGENT = "subagent"
    ACP = "acp"


class SessionState(Enum):
    ACTIVE = "active"
    IDLE = "idle"
    CLOSED = "closed"
    ERROR = "error"


@dataclass
class ThreadBinding:
    """线程绑定信息"""
    channel: str           # discord, telegram, dingtalk...
    channel_id: str        # 频道/群组 ID
    thread_id: Optional[str] = None  # 线程 ID（如果有）
    created_at: datetime = field(default_factory=datetime.now)


@dataclass
class Session:
    """会话实体"""
    key: str                              # agent:<agentId>:<runtime>:<uuid>
    agent_id: str
    runtime: RuntimeType
    state: SessionState = SessionState.ACTIVE
    thread_binding: Optional[ThreadBinding] = None
    created_at: datetime = field(default_factory=datetime.now)
    last_active_at: datetime = field(default_factory=datetime.now)
    metadata: Dict[str, Any] = field(default_factory=dict)
    
    @property
    def session_file(self) -> Path:
        """会话持久化文件路径"""
        return Path(f"sessions/{self.key}.jsonl")


class SessionManager:
    """
    会话管理器
    
    职责：
    1. 创建/恢复/关闭会话
    2. 管理线程绑定
    3. 会话持久化
    4. 事件分发
    """
    
    def __init__(self, sessions_dir: str = "sessions"):
        self.sessions_dir = Path(sessions_dir)
        self.sessions_dir.mkdir(exist_ok=True)
        
        # 内存中的会话注册表
        self._sessions: Dict[str, Session] = {}
        
        # 线程绑定索引 (channel:channel_id:thread_id -> session_key)
        self._thread_bindings: Dict[str, str] = {}
        
        # 生命周期钩子
        self._hooks: Dict[str, List[Callable]] = {
            "on_create": [],
            "on_resume": [],
            "on_close": [],
            "on_bind": [],
            "on_unbind": [],
        }
    
    def create_session(
        self,
        agent_id: str,
        runtime: RuntimeType,
        thread_binding: Optional[ThreadBinding] = None,
        metadata: Optional[Dict] = None,
    ) -> Session:
        """创建新会话"""
        session_key = f"agent:{agent_id}:{runtime.value}:{uuid.uuid4()}"
        
        session = Session(
            key=session_key,
            agent_id=agent_id,
            runtime=runtime,
            thread_binding=thread_binding,
            metadata=metadata or {},
        )
        
        # 注册会话
        self._sessions[session_key] = session
        
        # 处理线程绑定
        if thread_binding:
            self._bind_thread(session, thread_binding)
        
        # 持久化
        self._persist_session(session)
        
        # 触发钩子
        self._trigger_hooks("on_create", session)
        
        return session
    
    def get_session(self, session_key: str) -> Optional[Session]:
        """获取会话"""
        return self._sessions.get(session_key)
    
    def get_session_by_thread(
        self,
        channel: str,
        channel_id: str,
        thread_id: Optional[str] = None,
    ) -> Optional[Session]:
        """通过线程绑定获取会话"""
        binding_key = self._make_binding_key(channel, channel_id, thread_id)
        session_key = self._thread_bindings.get(binding_key)
        if session_key:
            return self._sessions.get(session_key)
        return None
    
    def close_session(self, session_key: str):
        """关闭会话"""
        session = self._sessions.get(session_key)
        if not session:
            return
        
        session.state = SessionState.CLOSED
        
        # 解除线程绑定
        if session.thread_binding:
            self._unbind_thread(session)
        
        # 触发钩子
        self._trigger_hooks("on_close", session)
    
    def bind_thread(
        self,
        session_key: str,
        thread_binding: ThreadBinding,
    ):
        """绑定会话到线程"""
        session = self._sessions.get(session_key)
        if not session:
            raise ValueError(f"Session not found: {session_key}")
        
        # 解除旧绑定
        if session.thread_binding:
            self._unbind_thread(session)
        
        # 新绑定
        self._bind_thread(session, thread_binding)
    
    def _bind_thread(self, session: Session, binding: ThreadBinding):
        """内部绑定实现"""
        binding_key = self._make_binding_key(
            binding.channel, 
            binding.channel_id, 
            binding.thread_id
        )
        
        self._thread_bindings[binding_key] = session.key
        session.thread_binding = binding
        
        self._trigger_hooks("on_bind", session, binding)
    
    def _unbind_thread(self, session: Session):
        """内部解绑实现"""
        if not session.thread_binding:
            return
        
        binding = session.thread_binding
        binding_key = self._make_binding_key(
            binding.channel,
            binding.channel_id,
            binding.thread_id,
        )
        
        self._thread_bindings.pop(binding_key, None)
        session.thread_binding = None
        
        self._trigger_hooks("on_unbind", session, binding)
    
    def _make_binding_key(
        self,
        channel: str,
        channel_id: str,
        thread_id: Optional[str],
    ) -> str:
        """生成绑定索引键"""
        if thread_id:
            return f"{channel}:{channel_id}:thread:{thread_id}"
        return f"{channel}:{channel_id}"
    
    def _persist_session(self, session: Session):
        """持久化会话"""
        session_file = self.sessions_dir / f"{session.key}.jsonl"
        
        # 追加会话创建事件
        event = {
            "type": "session_created",
            "key": session.key,
            "agent_id": session.agent_id,
            "runtime": session.runtime.value,
            "thread_binding": session.thread_binding.__dict__ if session.thread_binding else None,
            "created_at": session.created_at.isoformat(),
        }
        
        with open(session_file, "a") as f:
            f.write(json.dumps(event) + "\n")
    
    def _trigger_hooks(self, hook_name: str, *args):
        """触发生命周期钩子"""
        for hook in self._hooks.get(hook_name, []):
            hook(*args)
    
    def on(self, event: str, callback: Callable):
        """注册事件钩子"""
        if event in self._hooks:
            self._hooks[event].append(callback)


# 使用示例
manager = SessionManager()

# 创建会话
session = manager.create_session(
    agent_id="codex",
    runtime=RuntimeType.ACP,
    thread_binding=ThreadBinding(
        channel="discord",
        channel_id="123456789",
        thread_id="987654321",
    )
)

print(f"Created session: {session.key}")

# 通过线程查找会话
found = manager.get_session_by_thread(
    channel="discord",
    channel_id="123456789",
    thread_id="987654321",
)
print(f"Found by thread: {found.key if found else None}")

3.3 Runtime Router 设计

runtime_router.py - 运行时路由
"""
Runtime Router: 路由消息到正确的运行时
"""

from abc import ABC, abstractmethod
from typing import Dict, Type, Optional
from dataclasses import dataclass


@dataclass
class SpawnParams:
    """创建会话的参数"""
    task: str
    agent_id: str
    runtime: str = "subagent"
    mode: str = "run"          # run | session
    thread: bool = False
    cwd: Optional[str] = None
    model: Optional[str] = None
    label: Optional[str] = None


@dataclass
class SpawnResult:
    """创建会话的结果"""
    accepted: bool
    session_key: Optional[str] = None
    error: Optional[str] = None


class RuntimeBackend(ABC):
    """运行时后端抽象"""
    
    @property
    @abstractmethod
    def name(self) -> str:
        """运行时名称"""
        pass
    
    @abstractmethod
    async def spawn(self, params: SpawnParams) -> SpawnResult:
        """创建新会话"""
        pass
    
    @abstractmethod
    async def send(self, session_key: str, message: str):
        """发送消息到会话"""
        pass
    
    @abstractmethod
    async def close(self, session_key: str):
        """关闭会话"""
        pass
    
    @abstractmethod
    async def status(self, session_key: str) -> Dict:
        """获取会话状态"""
        pass


class ACPBackend(RuntimeBackend):
    """ACP 协议后端"""
    
    @property
    def name(self) -> str:
        return "acp"
    
    async def spawn(self, params: SpawnParams) -> SpawnResult:
        # 调用 ACP 客户端创建会话
        # 例如：启动 Claude Code / Codex
        import subprocess
        import os
        
        env = os.environ.copy()
        if params.cwd:
            env["WORKSPACE"] = params.cwd
        
        # 启动 ACP harness 进程
        process = subprocess.Popen(
            ["acpx", "start", params.agent_id],
            env=env,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
        )
        
        # 返回会话信息
        session_key = f"agent:{params.agent_id}:acp:{self._generate_uuid()}"
        
        return SpawnResult(
            accepted=True,
            session_key=session_key,
        )
    
    async def send(self, session_key: str, message: str):
        # 发送消息到 ACP session
        pass
    
    async def close(self, session_key: str):
        # 关闭 ACP session
        pass
    
    async def status(self, session_key: str) -> Dict:
        return {"status": "active"}


class SubagentBackend(RuntimeBackend):
    """Sub-agent 原生后端"""
    
    @property
    def name(self) -> str:
        return "subagent"
    
    async def spawn(self, params: SpawnParams) -> SpawnResult:
        # 在当前进程中创建子 Agent
        session_key = f"agent:{params.agent_id}:subagent:{self._generate_uuid()}"
        
        # 初始化子 Agent
        # agent = create_agent(params.agent_id, params.model)
        
        return SpawnResult(
            accepted=True,
            session_key=session_key,
        )
    
    async def send(self, session_key: str, message: str):
        pass
    
    async def close(self, session_key: str):
        pass
    
    async def status(self, session_key: str) -> Dict:
        return {"status": "active"}
    
    def _generate_uuid(self):
        import uuid
        return str(uuid.uuid4())


class RuntimeRouter:
    """
    运行时路由器
    
    根据请求参数选择正确的运行时后端
    """
    
    def __init__(self):
        self._backends: Dict[str, RuntimeBackend] = {}
        self._default_runtime = "subagent"
    
    def register_backend(self, backend: RuntimeBackend):
        """注册运行时后端"""
        self._backends[backend.name] = backend
    
    def get_backend(self, runtime: str) -> Optional[RuntimeBackend]:
        """获取运行时后端"""
        return self._backends.get(runtime)
    
    async def spawn(self, params: SpawnParams) -> SpawnResult:
        """创建会话"""
        runtime = params.runtime or self._default_runtime
        backend = self.get_backend(runtime)
        
        if not backend:
            return SpawnResult(
                accepted=False,
                error=f"Unknown runtime: {runtime}",
            )
        
        return await backend.spawn(params)
    
    async def route_message(
        self,
        session_key: str,
        message: str,
    ):
        """路由消息到会话"""
        # 从 session_key 解析 runtime
        parts = session_key.split(":")
        if len(parts) < 3:
            raise ValueError(f"Invalid session key: {session_key}")
        
        runtime = parts[2]  # agent:<id>:<runtime>:<uuid>
        backend = self.get_backend(runtime)
        
        if not backend:
            raise RuntimeError(f"No backend for runtime: {runtime}")
        
        await backend.send(session_key, message)


# 初始化路由器
router = RuntimeRouter()
router.register_backend(ACPBackend())
router.register_backend(SubagentBackend())

3.4 Thread Binding 实现

thread_binding.py - 线程绑定
"""
Thread Binding: 消息路由到绑定会话
"""

from dataclasses import dataclass
from typing import Dict, Optional, Callable, Awaitable
from session_manager import SessionManager, Session


@dataclass
class IncomingMessage:
    """入站消息"""
    channel: str              # discord, telegram...
    channel_id: str           # 频道/群组 ID
    thread_id: Optional[str]  # 线程 ID
    user_id: str
    content: str
    metadata: Dict = None


class ThreadBindingHandler:
    """
    线程绑定处理器
    
    职责：
    1. 检查消息是否来自绑定的线程
    2. 路由消息到正确的会话
    3. 处理绑定/解绑命令
    """
    
    def __init__(self, session_manager: SessionManager):
        self.session_manager = session_manager
        
        # 命令处理器
        self._commands: Dict[str, Callable] = {
            "/focus": self._handle_focus,
            "/unfocus": self._handle_unfocus,
            "/acp": self._handle_acp,
        }
    
    async def handle_message(
        self,
        message: IncomingMessage,
        send_reply: Callable[[str], Awaitable],
    ) -> bool:
        """
        处理入站消息
        
        Returns:
            True 如果消息已被处理（路由到绑定会话）
            False 如果消息未被处理
        """
        # 1. 检查是否是命令
        if message.content.startswith("/"):
            return await self._handle_command(message, send_reply)
        
        # 2. 查找线程绑定
        session = self.session_manager.get_session_by_thread(
            channel=message.channel,
            channel_id=message.channel_id,
            thread_id=message.thread_id,
        )
        
        if not session:
            # 没有绑定，不处理
            return False
        
        # 3. 路由到绑定会话
        # 实际实现中会调用 RuntimeRouter
        print(f"Routing message to session: {session.key}")
        
        return True
    
    async def _handle_command(
        self,
        message: IncomingMessage,
        send_reply: Callable[[str], Awaitable],
    ) -> bool:
        """处理命令"""
        parts = message.content.split()
        command = parts[0].lower()
        
        handler = self._commands.get(command)
        if handler:
            return await handler(message, parts[1:], send_reply)
        
        return False
    
    async def _handle_focus(
        self,
        message: IncomingMessage,
        args: list,
        send_reply: Callable[[str], Awaitable],
    ) -> bool:
        """
        /focus <session-label-or-key>
        
        将当前线程绑定到指定会话
        """
        if not args:
            await send_reply("用法: /focus <session-label-or-key>")
            return True
        
        target = args[0]
        
        # 查找会话
        session = self.session_manager.get_session(target)
        if not session:
            await send_reply(f"未找到会话: {target}")
            return True
        
        # 绑定
        from session_manager import ThreadBinding
        binding = ThreadBinding(
            channel=message.channel,
            channel_id=message.channel_id,
            thread_id=message.thread_id,
        )
        
        self.session_manager.bind_thread(session.key, binding)
        
        await send_reply(f"✅ 已绑定到会话: {session.key}")
        return True
    
    async def _handle_unfocus(
        self,
        message: IncomingMessage,
        args: list,
        send_reply: Callable[[str], Awaitable],
    ) -> bool:
        """
        /unfocus
        
        解除当前线程的绑定
        """
        session = self.session_manager.get_session_by_thread(
            channel=message.channel,
            channel_id=message.channel_id,
            thread_id=message.thread_id,
        )
        
        if not session:
            await send_reply("当前线程未绑定任何会话")
            return True
        
        self.session_manager.bind_thread(session.key, None)
        
        await send_reply(f"✅ 已解绑会话: {session.key}")
        return True
    
    async def _handle_acp(
        self,
        message: IncomingMessage,
        args: list,
        send_reply: Callable[[str], Awaitable],
    ) -> bool:
        """
        /acp spawn <agent> [--thread auto|here|off]
        
        创建 ACP 会话并绑定线程
        """
        if len(args) < 2 or args[0] != "spawn":
            await send_reply("用法: /acp spawn <agent> [--thread auto|here|off]")
            return True
        
        agent_id = args[1]
        thread_mode = "auto"  # default
        
        # 解析参数
        i = 2
        while i < len(args):
            if args[i] == "--thread" and i + 1 < len(args):
                thread_mode = args[i + 1]
                i += 2
            else:
                i += 1
        
        # 创建会话
        from runtime_router import RuntimeRouter, SpawnParams, RuntimeType
        
        params = SpawnParams(
            task="",  # 用户会在后续消息中发送
            agent_id=agent_id,
            runtime="acp",
            mode="session",
            thread=True,
        )
        
        # TODO: 调用 router.spawn()
        
        await send_reply(f"🚀 正在创建 {agent_id} 会话...")
        return True

harness.py - 完整示例
"""
完整的 Harness 实现
整合 Session Manager + Runtime Router + Thread Binding
"""

import asyncio
from typing import Optional, Callable, Awaitable
from dataclasses import dataclass

from session_manager import SessionManager, Session, RuntimeType, ThreadBinding
from runtime_router import RuntimeRouter, SpawnParams
from thread_binding import ThreadBindingHandler, IncomingMessage


@dataclass
class HarnessConfig:
    """Harness 配置"""
    sessions_dir: str = "sessions"
    default_runtime: str = "subagent"
    max_sessions: int = 100
    session_timeout: int = 3600  # 1 hour


class AgentHarness:
    """
    Agent Harness 主类
    
    整合所有组件，提供统一的 API
    """
    
    def __init__(self, config: Optional[HarnessConfig] = None):
        self.config = config or HarnessConfig()
        
        # 初始化组件
        self.session_manager = SessionManager(self.config.sessions_dir)
        self.runtime_router = RuntimeRouter()
        self.thread_handler = ThreadBindingHandler(self.session_manager)
        
        # 注册事件钩子
        self._setup_hooks()
    
    def _setup_hooks(self):
        """设置事件钩子"""
        @self.session_manager.on("on_create")
        def on_session_create(session: Session):
            print(f"[Harness] Session created: {session.key}")
        
        @self.session_manager.on("on_bind")
        def on_thread_bind(session: Session, binding: ThreadBinding):
            print(f"[Harness] Thread bound: {binding.channel}:{binding.thread_id}")
        
        @self.session_manager.on("on_close")
        def on_session_close(session: Session):
            print(f"[Harness] Session closed: {session.key}")
    
    async def spawn(
        self,
        agent_id: str,
        task: str,
        runtime: str = "subagent",
        thread: bool = False,
        mode: str = "run",
        **kwargs,
    ) -> Session:
        """
        创建新的 Agent 会话
        
        Args:
            agent_id: Agent 标识符
            task: 初始任务
            runtime: 运行时类型 (subagent/acp)
            thread: 是否绑定线程
            mode: 运行模式 (run/session)
        
        Returns:
            创建的 Session
        """
        # 创建 Session 记录
        runtime_type = RuntimeType.ACP if runtime == "acp" else RuntimeType.SUBAGENT
        
        thread_binding = None
        if thread and kwargs.get("thread_binding"):
            thread_binding = kwargs["thread_binding"]
        
        session = self.session_manager.create_session(
            agent_id=agent_id,
            runtime=runtime_type,
            thread_binding=thread_binding,
        )
        
        # 启动运行时
        params = SpawnParams(
            task=task,
            agent_id=agent_id,
            runtime=runtime,
            mode=mode,
            thread=thread,
            cwd=kwargs.get("cwd"),
            model=kwargs.get("model"),
        )
        
        result = await self.runtime_router.spawn(params)
        
        if not result.accepted:
            self.session_manager.close_session(session.key)
            raise RuntimeError(f"Failed to spawn: {result.error}")
        
        return session
    
    async def send(
        self,
        session_key: str,
        message: str,
    ):
        """发送消息到会话"""
        await self.runtime_router.route_message(session_key, message)
    
    async def handle_message(
        self,
        message: IncomingMessage,
        reply: Callable[[str], Awaitable],
    ) -> bool:
        """处理入站消息（检查线程绑定）"""
        return await self.thread_handler.handle_message(message, reply)
    
    def get_session(self, session_key: str) -> Optional[Session]:
        """获取会话"""
        return self.session_manager.get_session(session_key)
    
    def close_session(self, session_key: str):
        """关闭会话"""
        self.session_manager.close_session(session_key)
    
    def list_sessions(self) -> list:
        """列出所有活跃会话"""
        return list(self.session_manager._sessions.values())


# 使用示例
async def main():
    harness = AgentHarness()
    
    # 创建 ACP 会话
    session = await harness.spawn(
        agent_id="codex",
        task="分析这个仓库的架构",
        runtime="acp",
        mode="session",
        thread=True,
    )
    
    print(f"Created: {session.key}")
    
    # 发送消息
    await harness.send(session.key, "请继续")
    
    # 列出会话
    for s in harness.list_sessions():
        print(f"  - {s.key} ({s.state.value})")

if __name__ == "__main__":
    asyncio.run(main())
